import React from 'react';
import store from '../store';
import ProductMetadata from './ProductMetadata';
import MetadataTag from './common/MetadataTagComp';
import MVMComponent from './MVM/MVM';
import * as MetadataActions from '../action/metadata';


class metadataComponent extends React.Component{
    constructor(props) {
        super(props);
         this.handleSubmit = this.handleSubmit.bind(this);
         this.handleSubmit1 = this.handleSubmit1.bind(this);
         this.componentWillMount = this.componentWillMount.bind(this);
    }

    componentWillMount(){
      store.dispatch(MetadataActions.fetch_PMDData());
      store.dispatch(MetadataActions.fetch_MetaData());
      store.dispatch(MetadataActions.fetch_MVMData());
      store.dispatch(MetadataActions.fetch_QLTData());
      store.dispatch(MetadataActions.fetch_PName_Source());
    }

    handleSubmit() { 
        this.refs.productMetadata.submit();
    }

    handleSubmit1() { 
         this.refs.mvmForm.submit();
    }

    render() {
            return (
            	<form>
              <div className="pe-assetmetadata"> 
              <section>
                            <ProductMetadata 
                            ref="productMetadata"
                            onSubmit={this.handleSubmit1}
                            data={this.props.PMD} 
                            actions={this.props.actions}
                            pNameSource={this.props.PName_Source} 
                            handleChange={this.PMDChangeHandler}/>
                        </section>
              <section>
                          <MetadataTag heading="Product Level Tags"
                           actions={this.props.actions} 
                           suggestions={this.props.QLT.suggestions} 
                           tags={this.props.QLT.tags}/>
                        </section>
              <section>
                            <MetadataTag heading="Question Level Tags" 
                            actions={this.props.actions}  
                            suggestions={this.props.PLT.suggestions} 
                            tags={this.props.PLT.tags}
                            />
                        </section>
              <section>
                            <MVMComponent ref="mvmForm"
                            onSubmit={this.props.actions.MVMSaveHandler}
                              data={this.props.MVMData} 
                              actions={this.props.actions}> 
                            </MVMComponent>
                        </section>
              <section>
                        <button type="button" onClick={this.handleSubmit}>
                                <i/> Save
                        </button>
                        <button type="button" onClick={this.resetForm}>
                               Cancel
                        </button>
                    </section>
                </div>
                </form>
            )
    }
}

module.exports = metadataComponent;
