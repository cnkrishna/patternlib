import React from 'react';
import Label from './common/Label';
import TextBox from './common/TextBox';
import Heading from './common/Heading';
import SelectBox from './common/SelectBox';
import AutoComplete from './common/AutoComplete';
import {reduxForm} from 'redux-form';
import FormMessages from 'redux-form-validation';
import {generateValidation} from 'redux-form-validation';

var validations = {
     Author: false,
     'ISBN': false,
     'Publisher': false,
     'Disciplines': false
};


class ProductMetadataComponent extends React.Component{
  constructor(props) {
    super(props);
    this.displayName = 'ProductMetadata';
    this.handleChange = this.handleChange.bind(this);
    this.dataChange = this.dataChange.bind(this);
 }

handleChange(obj) {
  this.props.actions.PMDChangeHandler(obj);
}

dataChange(obj){
  this.props.actions.PMDChangeHandler({"id":"ProductName","value":obj.value});
  this.props.actions.fetch_New_PName_Source(obj.value);
}

render(){

  const{
    fields : {Author, ISBN, Publisher, Disciplines}
  }= this.props;

    	return(
    			<div>
    			 	     <div>
                        <Heading headingType="h3"  value="Product MetaData"/>
                 	</div>
                 	<div className="pe-input pe-input--horizontal">
                            <Label for="ProductName" text="Product Name(Book or Course Title)"/>
                            <AutoComplete id="ProductName" value={this.props.data.ProductName} suggestions={this.props.pNameSource} onSuggestionsUpdateRequested={this.dataChange}/>
                	</div >
                	<div className="pe-input pe-input--horizontal">
                        <Label for="Author" text="Author"/>
                        <TextBox id="Author" 
                        {...Author}
                        value ={this.props.data.Author}
                        handleChange={this.handleChange}/>
                        <FormMessages tagName="ul" errorCount="1" field={Author}>
                            <li when="required">
                                  Error:required field
                            </li>
                         </FormMessages>

                	</div >
                	<div className="pe-input pe-input--horizontal">
                        <Label for="ISBN" text="ISBN"/>
                        <TextBox id="ISBN"  
                        {...ISBN} 
                        value ={this.props.data.ISBN}
                        handleChange={this.handleChange}/>
                	</div >
                	<div className="pe-input pe-input--horizontal">
                          <Label for ="Publisher" text="Publisher"/>
                          <SelectBox id="Publisher" 
                          {...Publisher}
                           defaultData={this.props.data.Publisher} 
                           handleChange={this.handleChange} 
                           options={this.props.data.PublisherData}>
                          </SelectBox>
                	</div>
                    <div className="pe-input pe-input--horizontal">
                          <Label for ="Disciplines" text="Disciplines"/>
                          <SelectBox id="Disciplines" 
                          {...Disciplines}
                          defaultData={this.props.data.Disciplines} 
                          handleChange={this.handleChange} options={this.props.data.DisciplinesData}>
                          </SelectBox>
                	</div>
    			</div>

    		)
    }
};

ProductMetadataComponent= reduxForm(
  {
    form: 'productMetadata',
    fields:['Author', 'ISBN', 'Publisher', 'Disciplines'],
    ...generateValidation(validations)
  }
)(ProductMetadataComponent);

module.exports = ProductMetadataComponent;
